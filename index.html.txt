<html lang="en">
<head>
  <meta charset="UTF-8" />
  <meta name="viewport" content="width=device-width, initial-scale=1.0"/>
  <title>JARVIS HUD</title>
  <style>
    body, html { margin:0; padding:0; overflow:hidden; background:#000; }
    canvas { display:block; width:100vw; height:100vh; }
  </style>
</head>
<body>
  <canvas id="hud"></canvas>
  <script>
    const canvas = document.getElementById('hud');
    const ctx = canvas.getContext('2d');

    function resize() {
      canvas.width = window.innerWidth;
      canvas.height = window.innerHeight;
    }
    window.addEventListener('resize', resize);
    resize();

    function drawGrid() {
      const step = 50;
      ctx.strokeStyle = 'rgba(0,255,200,0.2)';
      ctx.lineWidth = 0.5;
      for (let x = 0; x < canvas.width; x += step) {
        for (let y = 0; y < canvas.height; y += step) {
          ctx.strokeRect(x, y, step, step);
        }
      }
    }

    function drawText() {
      const now = new Date();
      ctx.fillStyle = 'rgba(0,255,200,1)';
      ctx.font = '16px monospace';
      ctx.fillText(`DATE: ${now.toLocaleDateString('en-GB')}`, 20, 30);
      ctx.fillText(`TIME: ${now.toLocaleTimeString('en-US',{hour12:false})}`, 20, 60);

      ctx.fillText('SYSTEM METRICS', 20, 110);
      ctx.fillText('CPU Usage: 35%', 20, 135);
      ctx.fillText('RAM Usage: 58%', 20, 160);
      ctx.fillText('Network: 120 Mbps', 20, 185);

      ctx.fillText('NOTIFICATIONS', 20, 240);
      ctx.fillText('- Update available', 20, 265);
      ctx.fillText('- New email from Stark', 20, 290);
      ctx.fillText('- Battery low on Drone', 20, 315);

      ctx.fillText('-- JARVIS HUD v1.0', 20, canvas.height - 30);
    }

    function loop() {
      ctx.clearRect(0, 0, canvas.width, canvas.height);
      drawGrid();
      drawText();
      requestAnimationFrame(loop);
    }
    loop();
  </script>
</body>
</html>
